//二叉链表

struct ELemType
{
	int value;
};
typedef struct BiTNode
{
	ELemType data;
	struct BiTNode* lchild, * rchild;
}BiTNode,*BiTree;

//定义一颗空树

BiTree root = NULL;

//插入根节点
root = (BiTree)malloc(sizeof(BiTNode));
root->data = { 1 };
root->lchild = NULL;
root->rchild = NULL;

//插入新节点
BiTNode* p = (BiTNode*)malloc(sizeof(BiTNode));
p->data = { 2 };
p->lchild = NULL;
p->rchild = NULL;
root->lchild = p;//作为根节点的左孩子
